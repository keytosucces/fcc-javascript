function chunkArrayInGroups(arr, size) {
  let barr = [];
for (let i = 0; i < arr.length; i += size) {
  barr.push(arr.slice(i, i + size));
}
return barr;
}

console.log(chunkArrayInGroups(["a", "b", "c", "d"], 2));
